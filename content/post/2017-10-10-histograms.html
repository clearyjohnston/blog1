Histograms

library(sqldf)
library(Lahman)
library(ggplot2)
# extracting data----------------

query<-"SELECT weight
FROM Master"

result<-sqldf(query)

# visualizing data-------------

ggplot()+
  geom_histogram(data=result,aes(x=weight),color="green",fill="white",bins="50")+
  ggtitle("Body Weigh Distribution for Baseball Players")
